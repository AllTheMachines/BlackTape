Actually, the simplest approach: use std::process::Command to run a curl/powershell download. No — that's fragile. Let's add reqwest for the download only, it's needed.

      - FINAL APPROACH: Add `reqwest` with `stream` feature to Cargo.toml. Create download command:
        ```rust
        #[derive(Clone, serde::Serialize)]
        struct DownloadProgress { downloaded: u64, total: u64 }

        #[tauri::command]
        async fn download_model(
            app: tauri::AppHandle,
            url: String,
            filename: String,
            on_progress: tauri::ipc::Channel<DownloadProgress>
        ) -> Result<String, String> {
            let models_dir = app.path().app_data_dir().map_err(|e| e.to_string())?.join("models");
            std::fs::create_dir_all(&models_dir).map_err(|e| e.to_string())?;
            let target = models_dir.join(&filename);

            let client = reqwest::Client::new();
            let response = client.get(&url).send().await.map_err(|e| e.to_string())?;
            let total = response.content_length().unwrap_or(0);
            let mut downloaded: u64 = 0;
            let mut file = std::fs::File::create(&target).map_err(|e| e.to_string())?;

            use std::io::Write;
            use futures_util::StreamExt;
            let mut stream = response.bytes_stream();
            while let Some(chunk) = stream.next().await {
                let chunk = chunk.map_err(|e| e.to_string())?;
                file.write_all(&chunk).map_err(|e| e.to_string())?;
                downloaded += chunk.len() as u64;
                // Send progress every ~1MB to avoid IPC flood
                if downloaded % (1024 * 1024) < chunk.len() as u64 {
                    let _ = on_progress.send(DownloadProgress { downloaded, total });
                }
            }
            let _ = on_progress.send(DownloadProgress { downloaded, total });

            Ok(target.to_string_lossy().to_string())
        }
        ```

    ```typescript
    const MODELS = {
        generation: {
            url: 'https://huggingface.co/google/gemma-3-1b-it-qat-q4_0-gguf/resolve/main/gemma-3-1b-it-qat-q4_0.gguf',
            filename: 'gemma-3-1b-it-qat-q4_0.gguf',
            sizeBytes: 524_288_000,
            label: 'Text generation (~500MB)'
        },
        embedding: {
            url: 'https://huggingface.co/Qwen/Qwen3-Embedding-0.6B-GGUF/resolve/main/qwen3-embedding-0.6b-q5_k_m.gguf',
            filename: 'qwen3-embedding-0.6b-q5_k_m.gguf',
            sizeBytes: 440_000_000,
            label: 'Embeddings (~440MB)'
        }
    };
    export type ModelKey = keyof typeof MODELS;
    export { MODELS };

    export async function downloadModel(
        modelKey: ModelKey,
        onProgress: (downloaded: number, total: number) => void
    ): Promise<string> {
        const { invoke } = await import('@tauri-apps/api/core');
        const { Channel } = await import('@tauri-apps/api/core');
        const model = MODELS[modelKey];
        const channel = new Channel<{ downloaded: number; total: number }>();
        channel.onmessage = (msg) => onProgress(msg.downloaded, msg.total);
        return invoke('download_model', {
            url: model.url,
            filename: model.filename,
            onProgress: channel
        });
    }

    export async function checkModelExists(modelKey: ModelKey): Promise<boolean> {
        const { invoke } = await import('@tauri-apps/api/core');
        return invoke('check_model_exists', { filename: MODELS[modelKey].filename });
    }
    ```
