Create `src/lib/comms/dms.svelte.ts`:

This module manages encrypted 1:1 DMs using NIP-17 (gift-wrap) via NDK. Anti-pattern: NEVER use NIP-04 (kind:4) — NIP-04 leaks conversation graph. Use NDK's DM conversation API which handles NIP-17 automatically.

```typescript
import type NDK from '@nostr-dev-kit/ndk';
import { ndkState } from './nostr.svelte.js';
import { notifState } from './notifications.svelte.js';

// A single DM message in a conversation thread
export interface DmMessage {
  id: string;
  senderPubkey: string;
  content: string;
  createdAt: number;  // unix timestamp
  read: boolean;
}

// A conversation with one other user
export interface DmConversation {
  peerPubkey: string;         // the other party's pubkey
  messages: DmMessage[];
  lastMessageAt: number;
  unreadCount: number;
}

export const dmState = $state({
  conversations: [] as DmConversation[],
  loading: false,
});

// Send a DM using NIP-17 (gift-wrap) via NDK
// NDK's NDKDMConversation handles all NIP-17/NIP-59 complexity automatically
export async function sendDM(recipientPubkey: string, content: string): Promise<void> {
  const { ndk } = ndkState;
  if (!ndk) throw new Error('NDK not initialized');

  const { NDKUser, NDKDMConversation } = await import('@nostr-dev-kit/ndk');
  const recipient = new NDKUser({ pubkey: recipientPubkey });
  const conversation = new NDKDMConversation(ndk, [recipient]);
  await conversation.sendMessage(content);

  // Optimistically add to local state
  const existing = dmState.conversations.find(c => c.peerPubkey === recipientPubkey);
  const msg: DmMessage = {
    id: `local-${Date.now()}`,
    senderPubkey: ndkState.pubkey ?? '',
    content,
    createdAt: Math.floor(Date.now() / 1000),
    read: true,
  };
  if (existing) {
    existing.messages = [...existing.messages, msg];
    existing.lastMessageAt = msg.createdAt;
  } else {
    dmState.conversations = [
      ...dmState.conversations,
      { peerPubkey: recipientPubkey, messages: [msg], lastMessageAt: msg.createdAt, unreadCount: 0 }
    ];
  }
}

// Subscribe to incoming DMs for the current user
// Call once in root layout after initNostr() — stays alive for the session
export async function subscribeToIncomingDMs(): Promise<void> {
  const { ndk } = ndkState;
  if (!ndk || !ndkState.pubkey) return;

  const { NDKKind } = await import('@nostr-dev-kit/ndk');

  // NIP-17: incoming gift-wrapped DMs are kind:1059 addressed to our pubkey
  const sub = ndk.subscribe(
    { kinds: [1059 as NDKKind], '#p': [ndkState.pubkey] },
    { closeOnEose: false }
  );

  sub.on('event', async (event) => {
    // NDK can decrypt NIP-17 gift-wrap using our signer
    let decrypted: string;
    try {
      decrypted = await ndk.signer!.decrypt(event.pubkey, event.content);
    } catch {
      return; // not addressed to us or decryption failed
    }

    // Parse the inner sealed event to get sender pubkey and actual content
    let inner: { pubkey: string; content: string; created_at: number };
    try { inner = JSON.parse(decrypted); } catch { return; }

    const senderPubkey = inner.pubkey;
    const msg: DmMessage = {
      id: event.id,
      senderPubkey,
      content: inner.content,
      createdAt: inner.created_at,
      read: false,
    };

    const existing = dmState.conversations.find(c => c.peerPubkey === senderPubkey);
    if (existing) {
      // avoid duplicates
      if (!existing.messages.find(m => m.id === event.id)) {
        existing.messages = [...existing.messages, msg];
        existing.lastMessageAt = msg.createdAt;
        existing.unreadCount += 1;
      }
    } else {
      dmState.conversations = [
        ...dmState.conversations,
        { peerPubkey: senderPubkey, messages: [msg], lastMessageAt: msg.createdAt, unreadCount: 1 }
      ];
    }

    // Update global notification badge
    notifState.dmUnread = dmState.conversations.reduce((sum, c) => sum + c.unreadCount, 0);
  });
}

// Mark all messages in a conversation as read
export function markConversationRead(peerPubkey: string): void {
  const conv = dmState.conversations.find(c => c.peerPubkey === peerPubkey);
  if (conv) {
    conv.messages = conv.messages.map(m => ({ ...m, read: true }));
    conv.unreadCount = 0;
    notifState.dmUnread = dmState.conversations.reduce((sum, c) => sum + c.unreadCount, 0);
  }
}
```

Create `src/lib/comms/ai-taste-bridge.ts`:

This module implements ROADMAP success criteria 6 (AI taste translation) and 7 (AI matchmaking context). When a DM conversation thread is opened, it fetches the peer's taste profile from their public Mercury profile, then calls the user's configured AI provider to generate: (a) a "musical bridge" paragraph explaining WHY their tastes overlap, and (b) 2-3 conversation starter suggestions.

The result is stored in $state keyed by peerPubkey so it only generates once per conversation per session. Uses `getAiProvider()` from the existing AI module (Phase 9 pattern — same provider powers all AI features).

```typescript
import { ndkState } from './nostr.svelte.js';

// Bridge result for a single peer
export interface TasteBridgeResult {
  peerPubkey: string;
  bridgeExplanation: string;   // "Your tastes converge around post-punk atmosphere and electronic textures..."
  conversationStarters: string[];  // ["Have you heard X's latest?", ...]
  loading: boolean;
  error: string | null;
}

// Cache keyed by peerPubkey — generated once per session, stays alive until page reload
export const tasteBridgeState = $state(new Map<string, TasteBridgeResult>());

// Generate the AI taste bridge for a given peer
// Call this when a dm-thread view is opened for a peerPubkey
// Returns immediately (sets loading state); updates tasteBridgeState reactively when done
export async function getTasteBridge(peerPubkey: string): Promise<void> {
  // Already generated this session — skip
  if (tasteBridgeState.has(peerPubkey) && !tasteBridgeState.get(peerPubkey)!.loading) return;

  // Set loading state
  tasteBridgeState.set(peerPubkey, {
    peerPubkey,
    bridgeExplanation: '',
    conversationStarters: [],
    loading: true,
    error: null,
  });

  try {
    // Load the user's own taste profile (from taste.db via existing Tauri commands)
    // Dynamic import to avoid loading taste module before Tauri is ready
    const { tasteState } = await import('$lib/ai/state.svelte.js');

    // Fetch peer's public Mercury taste profile from Nostr
    // Per Phase 9 design: users publish their taste fingerprint as a kind:30078 event
    // with a 'd' tag of 'mercury-taste-profile'. We query for it here.
    const ndk = ndkState.ndk;
    let peerTasteProfile: string = 'unknown taste profile';

    if (ndk) {
      try {
        const { NDKEvent } = await import('@nostr-dev-kit/ndk');
        const events = await ndk.fetchEvents({
          kinds: [30078],
          authors: [peerPubkey],
          '#d': ['mercury-taste-profile'],
          limit: 1,
        });
        const profileEvent = [...events][0];
        if (profileEvent?.content) {
          // Content is JSON with { topTags: string[], topArtists: string[] }
          const profile = JSON.parse(profileEvent.content);
          const tags = (profile.topTags ?? []).slice(0, 8).join(', ');
          const artists = (profile.topArtists ?? []).slice(0, 5).join(', ');
          peerTasteProfile = `Tags: ${tags || 'none shared'}. Artists: ${artists || 'none shared'}.`;
        }
      } catch {
        // Peer hasn't published a taste profile — proceed with partial context
      }
    }

    // Build own taste summary from local state
    const ownTags = (tasteState?.topTags ?? []).slice(0, 8).join(', ') || 'varied taste';
    const ownArtists = (tasteState?.topArtists ?? []).slice(0, 5).join(', ') || 'various artists';
    const ownProfile = `Tags: ${ownTags}. Artists: ${ownArtists}.`;

    // Call the user's configured AI provider
    const { getAiProvider } = await import('$lib/ai/state.svelte.js');
    const provider = await getAiProvider();

    if (!provider) {
      // AI not configured — show a graceful fallback (no error, just no AI context)
      tasteBridgeState.set(peerPubkey, {
        peerPubkey,
        bridgeExplanation: 'Configure an AI model in Settings to see your musical connection.',
        conversationStarters: ['What have you been listening to lately?'],
        loading: false,
        error: null,
      });
      return;
    }

    const prompt = `You are Mercury's taste analyst. Given two music listener profiles, explain their musical connection and suggest conversation starters.

Your profile: ${ownProfile}
Their profile: ${peerTasteProfile}

Respond with a JSON object:
{
  "bridge": "1-2 sentences explaining the musical bridge between these tastes — be specific about shared aesthetics, not generic.",
  "starters": ["starter 1", "starter 2", "starter 3"]
}

Keep it conversational, music-nerd specific, and under 60 words for the bridge. Starters should be natural questions or observations, not generic openers.`;

    const response = await fetch(`${provider.baseUrl}/v1/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${provider.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: provider.model,
        messages: [{ role: 'user', content: prompt }],
        temperature: 0.7,
        max_tokens: 200,
        response_format: { type: 'json_object' },
      }),
    });

    if (!response.ok) throw new Error(`AI request failed: ${response.status}`);

    const data = await response.json();
    const content = data.choices?.[0]?.message?.content ?? '{}';
    let parsed: { bridge?: string; starters?: string[] } = {};
    try { parsed = JSON.parse(content); } catch { /* use defaults */ }

    tasteBridgeState.set(peerPubkey, {
      peerPubkey,
      bridgeExplanation: parsed.bridge ?? 'You share a musical sensibility — start talking to find out how.',
      conversationStarters: parsed.starters ?? ['What have you been into lately?'],
      loading: false,
      error: null,
    });
  } catch (e: unknown) {
    const msg = e instanceof Error ? e.message : 'Failed to generate taste context';
    tasteBridgeState.set(peerPubkey, {
      peerPubkey,
      bridgeExplanation: '',
      conversationStarters: [],
      loading: false,
      error: msg,
    });
  }
}
```
