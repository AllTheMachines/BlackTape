button, .btn {
  display: inline-flex;

  border-color: var(--b-3);

  width: 26px;
  height: 26px;
  padding: 0;
  justify-content: center;

  background: var(--acc-bg);
  border-color: var(--b-acc);
  color: var(--acc);

  background: var(--acc-bg-h);

  background: transparent;
  border-color: transparent;
  color: var(--t-3);

  background: var(--bg-4);
  color: var(--t-2);
  border-color: var(--b-2);
}

/* Small variant */
.btn-sm, button.sm {
  height: 22px;
  padding: 0 8px;
  font-size: 10px;
}

/* --- Inputs and Selects --- */
input:not([type="range"]):not([type="checkbox"]):not([type="radio"]),
select,
textarea {
  background: var(--bg-4);
  border: 1px solid var(--b-2);
  border-radius: var(--r);
  color: var(--t-1);
  font-size: 11px;
  font-family: inherit;
  outline: none;
  transition: border-color 0.1s;
}

input:not([type="range"]):not([type="checkbox"]):not([type="radio"]):focus,
select:focus,
textarea:focus {
  border-color: var(--acc);
}

textarea::placeholder {
  color: var(--t-3);
}

/* --- Badges / Source badges --- */
.badge, .source-badge, .nav-badge {

  background: var(--bg-4);

  border-radius: var(--r);
}

**NOTE on scope:** These global styles will inherit into Svelte scoped components. Be careful not to break existing special-case button styles (like icon-only play buttons, range inputs) — the selectors above use specific exclusions for range/checkbox/radio.
