```javascript
const WIKIDATA_SPARQL_URL = 'https://query.wikidata.org/sparql';
const USER_AGENT = 'BlackTape/0.3.0 (https://github.com/nicholasgasior/blacktape)';

async function fetchWikidataBatch(sparql) {
  const url = `${WIKIDATA_SPARQL_URL}?query=${encodeURIComponent(sparql)}`;
  try {
    const response = await fetch(url, {
      headers: {
        Accept: 'application/sparql-results+json',
        'User-Agent': USER_AGENT
      }
    });
    if (!response.ok) {
      console.warn(`[geocoding] Wikidata returned HTTP ${response.status}`);
      return [];
    }
    const data = await response.json();
    return data?.results?.bindings ?? [];
  } catch (err) {
    console.warn(`[geocoding] Wikidata unreachable: ${err.message}`);
    return [];
  }
}
```

Wikidata SPARQL property mapping:
- P434 = MusicBrainz artist ID (links Wikidata entity to MB artist)
- P19 = place of birth (city-level entity)
- P625 = coordinate location (on the place entity)
- P27 = country of citizenship (fallback for country centroid)
- geof:latitude(), geof:longitude() = extract from P625 coordinate value

**Step D — Batch loop (50 MBIDs per SPARQL query):**
```javascript
const BATCH_SIZE = 50;
const SLEEP_MS = 1100;
const updateStmt = db.prepare(
  "UPDATE artists SET city_lat = ?, city_lng = ?, city_precision = ? WHERE mbid = ?"
);

let processed = 0;
for (let i = 0; i < artists.length; i += BATCH_SIZE) {
  const batch = artists.slice(i, i + BATCH_SIZE);
  const mbids = batch.map(a => a.mbid);

  const results = await fetchWikidataBatch(mbids);

  // Build a map: mbid -> { lat, lng, precision }
  // SPARQL returns one row per mbid (with SAMPLE() aggregation)
  // city precision wins over country if both are present (query uses OPTIONAL blocks)
  const geoMap = {};
  for (const row of results) {
    const mbid = row.mbid?.value;
    if (!mbid) continue;
    const lat = parseFloat(row.lat?.value);
    const lng = parseFloat(row.lng?.value);
    const precision = row.precisionVal?.value;
    if (!isNaN(lat) && !isNaN(lng) && precision) {
      // city beats country — only overwrite if not already set to 'city'
      if (!geoMap[mbid] || geoMap[mbid].precision === 'country') {
        geoMap[mbid] = { lat, lng, precision };
      }
    }
  }

  // Write results to DB in a single transaction
  const writeMany = db.transaction((entries) => {
    for (const [mbid, geo] of entries) {
      updateStmt.run(geo.lat, geo.lng, geo.precision, mbid);
    }
  });
  writeMany(Object.entries(geoMap));

  // Mark artists with no Wikidata result as 'no_data' so they are not retried
  // Use a placeholder precision value 'none' to mark them as checked
  const noDataMbids = mbids.filter(m => !geoMap[m]);
  const markNone = db.transaction((mbidList) => {
    for (const mbid of mbidList) {
      db.prepare("UPDATE artists SET city_precision = 'none' WHERE mbid = ? AND city_precision IS NULL").run(mbid);
    }
  });
  markNone(noDataMbids);

  processed += batch.length;
  if (processed % 1000 === 0 || i + BATCH_SIZE >= artists.length) {
    console.log(`[geocoding] ${processed}/${artists.length} processed, ${Object.keys(geoMap).length} geocoded in last batch`);
  }

  // Rate limit: 1100ms between Wikidata requests
  if (i + BATCH_SIZE < artists.length) {
    await new Promise(r => setTimeout(r, SLEEP_MS));
  }
}
```

**The SPARQL query function `fetchWikidataBatch(mbids)`:**

Use the precision-hierarchy query from RESEARCH.md:
```javascript
async function fetchWikidataBatch(mbids) {
  const valuesClause = mbids.map(id => `"${id}"`).join(' ');
  const sparql = `
    SELECT ?mbid (SAMPLE(?lat) AS ?lat) (SAMPLE(?lng) AS ?lng)
           (SAMPLE(?precisionVal) AS ?precisionVal)
    WHERE {
      VALUES ?mbid { ${valuesClause} }
      ?artist wdt:P434 ?mbid .
      OPTIONAL {
        ?artist wdt:P19 ?birthCity .
        ?birthCity wdt:P625 ?cityCoord .
        BIND(geof:latitude(?cityCoord) AS ?lat)
        BIND(geof:longitude(?cityCoord) AS ?lng)
        BIND("city" AS ?precisionVal)
      }
      OPTIONAL {
        ?artist wdt:P27 ?country .
        ?country wdt:P625 ?countryCoord .
        BIND(geof:latitude(?countryCoord) AS ?lat)
        BIND(geof:longitude(?countryCoord) AS ?lng)
        BIND("country" AS ?precisionVal)
      }
    }
    GROUP BY ?mbid
  `;
  // ... fetch with User-Agent header, return bindings array
}
```

**Critical implementation notes:**
- The precision value `'none'` is used as a sentinel to skip artists with no Wikidata match on re-runs. The `getGeocodedArtists()` query in Plan 04 must filter `WHERE city_precision IN ('city', 'country')` — this naturally excludes `'none'` rows.
- The LIMIT 500000 in Step C allows partial runs: remove the LIMIT for full 2.6M geocoding run.
- Sleep 1100ms between batches (not after the last batch) to stay under Wikidata rate limit.
- Script must be `async` at the top level since it uses `await` in the loop — use an IIFE: `(async () => { ... })();`
