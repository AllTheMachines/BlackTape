    ```typescript
    /**
     * Generates iframe and script-tag embed snippets for Mercury artist/collection embeds.
     * Used on artist page and collection page to give bloggers copy-paste code.
     */

    export interface EmbedSnippets {
      iframe: string;
      scriptTag: string;
    }

    export function generateEmbedSnippets(embedUrl: string, title: string): EmbedSnippets {
      const escaped = title.replace(/"/g, '&quot;');
      const iframe = `<iframe
      src="${embedUrl}"
      width="400"
      height="180"
      title="${escaped}"
      frameborder="0"
      style="border-radius:8px;overflow:hidden;max-width:100%;"
      loading="lazy"
    ></iframe>`;

      const scriptTag = `<div id="mercury-embed"
      data-src="${embedUrl}"
      data-title="${escaped}"
    ></div>
    <script src="${embedUrl.split('/embed/')[0]}/embed.js" async><\/script>`;

      return { iframe, scriptTag };
    }
    ```

    **src/lib/curator/qr.ts**
    ```typescript
    /**
     * QR code generation for Mercury embed URLs and collection URLs.
     * Uses qrcode npm package for SVG output — no canvas dependency.
     * Only called client-side (lazy import to avoid SSR issues).
     */

    export async function generateQrSvg(url: string, dark = false): Promise<string> {
      // Dynamic import to avoid SSR bundling issues
      const QRCode = (await import('qrcode')).default;
      return QRCode.toString(url, {
        type: 'svg',
        margin: 2,
        color: {
          dark: dark ? '#ffffff' : '#000000',
          light: dark ? '#1a1a1a' : '#ffffff',
        },
      });
    }
    ```

    **src/routes/embed.js/+server.ts**
    SvelteKit route named `embed.js` (directory `src/routes/embed.js/`) that serves a JavaScript file at `GET /embed.js`.

    The script does four things when loaded on an external page:
    1. Find the `<div id="mercury-embed" data-src="..." data-title="..." [data-curator="..."]>` div
    2. Read `data-src` (the embed URL), `data-title`, and optional `data-curator`
    3. Inject an `<iframe>` into the div using the data-src URL (same dimensions as the iframe snippet: 400×200, border-radius 8px)
    4. If `data-curator` is present, fire an attribution ping: parse the artist MBID from the embed URL path (`/embed/artist/[slug]`), then GET `/api/curator-feature?artist=[mbid-from-url]&curator=[handle]` — but since we only have the slug (not the MBID) in the embed URL, fire the ping to `/api/curator-feature?slug=[slug]&curator=[handle]` and handle slug-based lookup server-side (update Plan 12-03 Task 1 to accept `slug` as alternative to `artist` param), OR simpler: fire `[origin]/api/curator-feature?embed_url=[encodeURIComponent(data-src)]&curator=[handle]` and let the server extract the slug. **Simplest correct approach:** derive the Mercury origin from `data-src` (split on `/embed/`), then fire `${origin}/api/curator-feature?slug=${slug}&curator=${handle}` where slug is extracted from the embed URL path segment after `/embed/artist/`.

    The GET handler returns the JavaScript as `text/javascript`:

    ```typescript
    export function GET(): Response {
      const js = `(function() {
        var div = document.getElementById('mercury-embed');
        if (!div) return;
        var src = div.getAttribute('data-src');
        var title = div.getAttribute('data-title') || 'Mercury artist';
        var curator = div.getAttribute('data-curator');
        if (!src) return;
        // Inject iframe
        var iframe = document.createElement('iframe');
        iframe.src = src;
        iframe.width = '400';
        iframe.height = '200';
        iframe.title = title;
        iframe.frameBorder = '0';
        iframe.loading = 'lazy';
        iframe.style.cssText = 'border-radius:8px;overflow:hidden;max-width:100%;display:block;';
        div.parentNode.replaceChild(iframe, div);
        // Fire attribution ping if curator handle provided
        if (curator) {
          try {
            var parts = src.split('/embed/artist/');
            if (parts.length === 2) {
              var origin = parts[0];
              var slug = parts[1].split('?')[0].split('/')[0];
              var ping = origin + '/api/curator-feature?slug=' + encodeURIComponent(slug) + '&curator=' + encodeURIComponent(curator);
              fetch(ping, { mode: 'no-cors' }).catch(function() {});
            }
          } catch(e) {}
        }
      })();`;

      return new Response(js, {
        headers: {
          'Content-Type': 'text/javascript; charset=utf-8',
          'Cache-Control': 'public, max-age=86400',
          'Access-Control-Allow-Origin': '*',
        },
      });
    }
    ```

When `slug` param is provided instead of `artist`, look up the artist MBID: `SELECT mbid FROM artists WHERE slug = ? LIMIT 1`. Validate slug: alphanumeric + hyphens only, max 100 chars. Then proceed with MBID-based INSERT. Update the action in Plan 12-03 Task 1 to include this slug-based lookup path.

    Note: The file is at `src/routes/embed.js/+server.ts`. SvelteKit serves routes based on directory name — the directory `embed.js` results in the URL path `/embed.js`. This is the correct pattern for serving static-ish JS files via SvelteKit routes.
