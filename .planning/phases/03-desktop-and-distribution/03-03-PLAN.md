  <name>Task 1: Create platform detection and universal search load function</name>
  <files>
    src/lib/platform.ts
    src/routes/search/+page.ts

  <action>
    **src/lib/platform.ts:**
    Create a simple platform detection utility:
    ```typescript
    export function isTauri(): boolean {
      return typeof window !== 'undefined' &&
        (window as any).__TAURI_INTERNALS__ !== undefined;
    }
    ```
    This works because Tauri 2.x injects `__TAURI_INTERNALS__` into the webview's window object. This check is reliable and doesn't require importing from @tauri-apps/api (which would fail in the web build).

    **src/routes/search/+page.ts:**
    Create a universal load function. This file coexists with +page.server.ts:
    - In the web build (SSR): SvelteKit runs +page.server.ts on the server, then passes its return value as the `data` prop to +page.ts. The universal load function receives this via `data` parameter and passes it through.
    - In the desktop build (SPA, ssr=false): +page.server.ts is not executed. The universal load function detects Tauri and queries the local database directly.

    Implementation:
    ```typescript
    import { isTauri } from '$lib/platform';

    export const load = async ({ url, data }) => {
      // Web SSR: data comes from +page.server.ts
      if (!isTauri()) {
        return data;
      }

      // Tauri desktop: query local database
      const { getProvider } = await import('$lib/db/provider');
      const { searchArtists, searchByTag } = await import('$lib/db/queries');

      const q = url.searchParams.get('q')?.trim() ?? '';
      const mode = url.searchParams.get('mode') === 'tag' ? 'tag' : 'artist';

      if (!q) {
        return { results: [], query: '', mode, matchedTag: null, error: false };
      }

      try {
        const provider = await getProvider();
        const results = mode === 'tag'
          ? await searchByTag(provider, q)
          : await searchArtists(provider, q);
        return { results, query: q, mode, matchedTag: mode === 'tag' ? q : null, error: false };
      } catch (err) {
        console.error('Search error:', err);
        return { results: [], query: q, mode, matchedTag: null, error: true };
      }
    };
    ```

    Use dynamic imports for `$lib/db/provider` and `$lib/db/queries` to avoid pulling Tauri dependencies into the web bundle. The web path returns early before any dynamic imports execute.
  </action>

  </verify>

<task type="auto">
  <name>Task 2: Create universal artist page load function</name>
  <files>
    src/routes/artist/[slug]/+page.ts
  </files>
  <action>
    Create a universal load function for the artist page. This is more complex than search because the artist page also fetches external data (links, releases, bio) from MusicBrainz via server API routes.

    In the web build: data comes from +page.server.ts (which calls internal /api/ routes).
    In the Tauri build: artist data comes from local DB. Links, releases, and bio are fetched directly from external APIs (no CORS issue in Tauri webview -- or if there is, graceful degradation).

    Implementation:
    ```typescript
    import { isTauri } from '$lib/platform';
    import { error } from '@sveltejs/kit';

    export const load = async ({ params, data, fetch }) => {
      // Web SSR: data comes from +page.server.ts
      if (!isTauri()) {
        return data;
      }

      // Tauri desktop: query local database + fetch external data
      const { getProvider } = await import('$lib/db/provider');
      const { getArtistBySlug } = await import('$lib/db/queries');

      const { slug } = params;
      const provider = await getProvider();
      const artist = await getArtistBySlug(provider, slug);

      if (!artist) {
        throw error(404, 'Artist not found');
      }

      // Fetch external data from MusicBrainz directly
      // In Tauri, we can call external APIs from the client
      // Use the same logic as the server API routes but called directly
      let links = { bandcamp: [], spotify: [], soundcloud: [], youtube: [], wikipedia: [], other: [] };
      let categorizedLinks = { streaming: [], official: [], social: [], info: [], support: [], other: [] };
      let releases = [];
      let bio = null;

      try {
        // Fetch links from MusicBrainz
        const mbLinksResponse = await fetch(
          `https://musicbrainz.org/ws/2/artist/${artist.mbid}?inc=url-rels&fmt=json`,
          {
            headers: {
              'User-Agent': 'Mercury/0.1.0 (https://github.com/user/mercury)',
              'Accept': 'application/json'
            }
          }
        );

        if (mbLinksResponse.ok) {
          const mbData = await mbLinksResponse.json();
          // Process links using the same categorization logic from $lib/embeds/categorize
          const { categorizeByRelationType, detectPlatform, labelFromUrl, emptyCategorizedLinks } =
            await import('$lib/embeds/categorize');

          categorizedLinks = emptyCategorizedLinks();
          const seen = new Set();

          if (mbData.relations) {
            for (const rel of mbData.relations) {
              if (rel['target-type'] === 'url' && rel.url?.resource) {
                const url = rel.url.resource;
                if (seen.has(url)) continue;
                seen.add(url);

                // Legacy format
                const hostname = new URL(url).hostname;
                if (hostname.includes('bandcamp.com')) links.bandcamp.push(url);
                else if (hostname.includes('open.spotify.com') || hostname === 'spotify.com') links.spotify.push(url);
                else if (hostname.includes('soundcloud.com')) links.soundcloud.push(url);
                else if (hostname.includes('youtube.com') || hostname === 'youtu.be') links.youtube.push(url);
                else if (hostname.includes('wikipedia.org')) links.wikipedia.push(url);
                else links.other.push(url);

                // Categorized format
                const category = rel.type ? categorizeByRelationType(rel.type) : 'other';
                const platform = detectPlatform(url);
                const finalCategory = platform ? 'streaming' : category;
                categorizedLinks[finalCategory].push({ url, label: labelFromUrl(url) });
              }
            }
          }
        }

        // Fetch releases from MusicBrainz
        const mbReleasesResponse = await fetch(
          `https://musicbrainz.org/ws/2/artist/${artist.mbid}?inc=release-groups&fmt=json`,
          {
            headers: {
              'User-Agent': 'Mercury/0.1.0 (https://github.com/user/mercury)',
              'Accept': 'application/json'
            }
          }
        );

        if (mbReleasesResponse.ok) {
          const mbRelData = await mbReleasesResponse.json();
          if (mbRelData['release-groups']) {
            releases = mbRelData['release-groups'].map((rg: any) => ({
              mbid: rg.id,
              title: rg.title,
              type: rg['primary-type'] || 'Other',
              year: rg['first-release-date']?.substring(0, 4) || null,
              coverArtUrl: `https://coverartarchive.org/release-group/${rg.id}/front-250`
            }));
          }
        }

        // Fetch Wikipedia bio if we have a Wikipedia link
        if (links.wikipedia.length > 0) {
          const { fetchWikipediaBio } = await import('$lib/bio');
          bio = await fetchWikipediaBio(links.wikipedia[0]);
        }
      } catch (err) {
        // External data is best-effort -- page renders from DB data alone
        console.error('External data fetch error:', err);
      }

      return { artist, links, categorizedLinks, releases, bio };
    };
    ```

    IMPORTANT NOTES:
    - Use `fetch` from the load function parameter -- SvelteKit provides this in both contexts.
    - MusicBrainz rate limiting: in the Tauri context, there's no module-level rate limiter like the server routes have. For now, this is acceptable because artist page loads are infrequent (user clicks, not automated). If it becomes an issue, add client-side rate limiting later.
    - The release data processing should match the structure expected by the artist page Svelte component. Check the existing API route at `src/routes/api/artist/[mbid]/releases/+server.ts` to understand the exact shape of the `ReleaseGroup` type.
    - Import types from `$lib/embeds/types` for type safety where possible.
    - The links and categorizedLinks objects must match the shapes expected by `+page.svelte` (check the component's `data` usage).
    - All external API failures are caught and the page renders with just the DB data (artist name, tags, country) -- this matches the existing "bio and links are best-effort" decision.
  </action>
