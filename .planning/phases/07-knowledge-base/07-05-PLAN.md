**`src/routes/api/time-machine/+server.ts`** — API endpoint for web client-side year/tag changes. NOT used by Tauri (Tauri queries DB directly):

```typescript
import { json } from '@sveltejs/kit';
import type { RequestHandler } from './$types';
import { D1Provider } from '$lib/db/d1-provider';
import { getArtistsByYear } from '$lib/db/queries';

export const GET: RequestHandler = async ({ url, platform }) => {
  const year = parseInt(url.searchParams.get('year') ?? '1990', 10);
  const tag = url.searchParams.get('tag') ?? undefined;

  if (isNaN(year) || year < 1900 || year > 2030) {
    return json({ artists: [], year }, { status: 400 });
  }

  const db = new D1Provider(platform!.env.DB);
  const artists = await getArtistsByYear(db, year, tag, 50);
  return json({ artists, year });
};
```

Note: D1Provider is imported from `$lib/db/d1-provider` (filename: `d1-provider.ts`, not `D1Provider.ts`).

Create `src/lib/components/GenreGraphEvolution.svelte` — a D3 force-directed graph that shows genre nodes whose `inception_year` is <= the current year, animating as years change. This is the "animated genre graph evolution" view from CONTEXT.md.

**Props:**
```typescript
let { currentYear, allNodes, allEdges }:
  { currentYear: number; allNodes: GenreNode[]; allEdges: GenreEdge[] } = $props();
```

**Filtering:**
- Only show genre nodes where `node.inception_year == null || node.inception_year <= currentYear`
- Edges: only include edges where both source and target nodes are visible
- When `currentYear` prop changes, recompute filtered nodes/edges and re-run simulation

**Animation behavior (Claude's discretion):**
- Use headless `simulation.tick(200)` pattern (same as GenreGraph.svelte, NOT on('tick'))
- When new nodes appear (year increases), they should "fade in" via CSS opacity transition — set initial opacity 0 on new nodes, transition to 1 over 400ms
- Nodes that newly appear get a brief scale-up animation (CSS transform: scale(0.5) → scale(1))
- Use a `$derived` or reactive block watching `currentYear` to trigger re-layout

**Visual design (Claude's discretion):**
- Node color encodes genre type: genre = primary, scene = blue, city = green (match GenreGraph.svelte colors)
- Node size scales with connectivity (number of edges)
- Label: show node.name for nodes with >=3 connections, hide for sparse nodes (reduces hairball effect)
- Show a subtle "year" watermark text in the graph background (large, low-opacity)

**Layout:**
- ViewBox="0 0 800 400", responsive 100% width
- Compact height (400px) — sits above the artist list in the Time Machine layout

**Performance note:** Re-running D3 simulation on every year change is intentional — the graph is small (<<200 nodes) and the headless tick pattern is fast. Do NOT attempt incremental layout updates.

**Types:** Import `GenreNode`, `GenreEdge` from `$lib/db/queries`.

```typescript
// Reactive derived state — recomputes when currentYear changes
let visibleNodes = $derived(
  allNodes.filter(n => n.inception_year == null || n.inception_year <= currentYear)
);
let visibleNodeSlugs = $derived(new Set(visibleNodes.map(n => n.slug)));
let visibleEdges = $derived(
  allEdges.filter(e => visibleNodeSlugs.has(e.source) && visibleNodeSlugs.has(e.target))
);

// Re-run layout whenever visible nodes change
let layout = $state<{ nodes: LayoutNode[]; edges: LayoutEdge[] }>({ nodes: [], edges: [] });
$effect(() => {
  // Run headless simulation and assign to layout state once settled
  const nodes = visibleNodes.map(n => ({ ...n, x: Math.random() * 800, y: Math.random() * 400 }));
  // ... D3 simulation.tick(200).stop() pattern (same as GenreGraph/StyleMap)
  layout = computeLayout(nodes, visibleEdges);
});
```

  <action>
Create `src/routes/time-machine/+page.svelte` with all three views per CONTEXT.md: animated genre graph evolution, year snapshot heading, and filtered artist/release list.

**CRITICAL: loadYear() must branch on isTauri() — the `/api/time-machine` endpoint does not exist in the Tauri static build (adapter-static has no server). Mirror the pattern from `src/routes/crate/+page.svelte`.**

```svelte
<script lang="ts">
  import type { PageData } from './$types';
  import ArtistCard from '$lib/components/ArtistCard.svelte';
  import GenreGraphEvolution from '$lib/components/GenreGraphEvolution.svelte';
  import { isTauri } from '$lib/platform';

  let { data }: { data: PageData } = $props();

  // --- State ---
  let currentYear = $state(data.year);
  let tagFilter = $state('');
  let artists = $state(data.artists);
  let loading = $state(false);

  // Genre graph data — loaded once on mount (all genres, filtered client-side by year)
  let allGenreNodes = $state<any[]>([]);
  let allGenreEdges = $state<any[]>([]);

  // Decade definitions
  const DECADES = [
    { label: '60s', start: 1960, end: 1969 },
    { label: '70s', start: 1970, end: 1979 },
    { label: '80s', start: 1980, end: 1989 },
    { label: '90s', start: 1990, end: 1999 },
    { label: '00s', start: 2000, end: 2009 },
    { label: '10s', start: 2010, end: 2019 },
    { label: '20s', start: 2020, end: 2029 },
  ];

  let activeDecade = $state(DECADES.find(d => currentYear >= d.start && currentYear <= d.end) ?? DECADES[3]);

  // --- Fetch artists on year/tag change ---
  // CRITICAL: Tauri has no server — fetch() to /api/time-machine would silently fail.
  // Tauri path: dynamic import provider + direct DB query (mirror crate/+page.svelte pattern).
  // Web path: fetch /api/time-machine.
  async function loadYear(year: number, tag?: string) {
    loading = true;
    try {
      if (isTauri()) {
        // Tauri: direct DB query — no fetch (adapter-static has no server)
        const { getProvider } = await import('$lib/db/provider');
        const { getArtistsByYear } = await import('$lib/db/queries');
        const db = await getProvider();
        artists = await getArtistsByYear(db, year, tag, 50);
        currentYear = year;
      } else {
        // Web: API fetch
        const params = new URLSearchParams({ year: String(year) });
        if (tag) params.set('tag', tag);
        const resp = await fetch(`/api/time-machine?${params}`);
        if (resp.ok) {
          const result = await resp.json();
          artists = result.artists;
          currentYear = year;
        }
      }
    } catch { /* best-effort */ } finally {
      loading = false;
    }
  }

  function selectDecade(decade: typeof DECADES[0]) {
    activeDecade = decade;
    const mid = Math.floor((decade.start + decade.end) / 2);
    currentYear = mid;
    loadYear(mid, tagFilter || undefined);
  }

  let slideTimer: ReturnType<typeof setTimeout>;
  let tagTimer: ReturnType<typeof setTimeout>;

  function onYearSlide(e: Event) {
    const year = parseInt((e.target as HTMLInputElement).value, 10);
    currentYear = year;
    clearTimeout(slideTimer);
    slideTimer = setTimeout(() => loadYear(year, tagFilter || undefined), 300);
  }

  function onTagFilter(e: Event) {
    tagFilter = (e.target as HTMLInputElement).value;
    clearTimeout(tagTimer);
    tagTimer = setTimeout(() => loadYear(currentYear, tagFilter || undefined), 500);
  }

  // Load genre graph data on mount for the evolution view
  import { onMount } from 'svelte';
  onMount(async () => {
    try {
      if (isTauri()) {
        const { getProvider } = await import('$lib/db/provider');
        const { getAllGenreGraph } = await import('$lib/db/queries');
        const db = await getProvider();
        const graph = await getAllGenreGraph(db);
        allGenreNodes = graph.nodes;
        allGenreEdges = graph.edges;
      } else {
        // Web: fetch all genres for the evolution graph
        const resp = await fetch('/api/genres');
        if (resp.ok) {
          const graph = await resp.json();
          allGenreNodes = graph.nodes ?? [];
          allGenreEdges = graph.edges ?? [];
        }
      }
    } catch { /* graph evolution is best-effort */ }
  });
</script>
```
