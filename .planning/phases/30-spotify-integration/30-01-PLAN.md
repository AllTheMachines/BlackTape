// src/lib/spotify/state.svelte.ts
export interface SpotifyStoredState {
  accessToken: string;
  refreshToken: string;
  tokenExpiry: number;  // Unix ms
  clientId: string;
  displayName: string;
}

export const spotifyState = $state({
  connected: false,
  displayName: '' as string,
  accessToken: '' as string,
  refreshToken: '' as string,
  tokenExpiry: 0 as number,
  clientId: '' as string
});

export function setSpotifyConnected(data: SpotifyStoredState): void {
  spotifyState.connected = true;
  spotifyState.displayName = data.displayName;
  spotifyState.accessToken = data.accessToken;
  spotifyState.refreshToken = data.refreshToken;
  spotifyState.tokenExpiry = data.tokenExpiry;
  spotifyState.clientId = data.clientId;
}

export function clearSpotifyState(): void {
  spotifyState.connected = false;
  spotifyState.displayName = '';
  spotifyState.accessToken = '';
  spotifyState.refreshToken = '';
  spotifyState.tokenExpiry = 0;
  spotifyState.clientId = '';
}
```

Key functions to implement:

**startSpotifyAuth(clientId: string): Promise<SpotifyStoredState>**
- Call start() from tauri-plugin-oauth to get port
- Build redirectUri = `http://127.0.0.1:${port}/callback`
- Build auth URL with scopes: 'user-read-private user-read-playback-state user-modify-playback-state'
- Include CSRF state param (generateRandomString(16))
- Open in system browser via open() from @tauri-apps/plugin-shell
- Await callback via onUrl() with 120s timeout (call cancel(port) on timeout, throw Error('Authorization timed out'))
- Exchange code for tokens via POST https://accounts.spotify.com/api/token
- Fetch display name via GET https://api.spotify.com/v1/me (field: display_name)
- Return SpotifyStoredState with: accessToken, refreshToken, tokenExpiry (Date.now() + expires_in * 1000), clientId, displayName

**saveSpotifyTokens(data: SpotifyStoredState): Promise<void>**
- Writes all 5 ai_settings keys: spotify_client_id, spotify_access_token, spotify_refresh_token, spotify_token_expiry (as string), spotify_display_name

**loadSpotifyState(): Promise<SpotifyStoredState | null>**

**clearSpotifyTokens(): Promise<void>**

// src/lib/spotify/api.ts

export class SpotifyAuthError extends Error {
  constructor(message = 'Token expired') { super(message); this.name = 'SpotifyAuthError'; }
}
export class SpotifyNotFoundError extends Error {
  constructor(message = 'Artist not found on Spotify') { super(message); this.name = 'SpotifyNotFoundError'; }
}

export type PlayResult = 'ok' | 'no_device' | 'premium_required' | 'token_expired';

/**
 * Extract Spotify artist ID from MusicBrainz Spotify URL.
 * Input: "https://open.spotify.com/artist/4Z8W4fohXX484ULPew5ay1"
 * Returns: "4Z8W4fohXX484ULPew5ay1" or null
 */
export function extractSpotifyArtistId(spotifyUrl: string): string | null {
  const match = spotifyUrl.match(/open\.spotify\.com\/artist\/([a-zA-Z0-9]+)/);
  return match ? match[1] : null;
}

/**
 * Get top track URIs for a Spotify artist (up to 10).
 * Throws SpotifyAuthError on 401, SpotifyNotFoundError on 404.
 */
export async function getArtistTopTracks(
  spotifyArtistId: string,
  accessToken: string
): Promise<string[]>;

/**
 * Trigger playback of track URIs on the user's active Spotify device.
 * Returns discriminated result — never throws.
 *
 * 'no_device' means: 404 with NO_ACTIVE_DEVICE or any unexpected error
 * Show message: "Open Spotify Desktop and start playing anything, then try again."
 *
 * 'premium_required' means: 403
 * Show message: "Spotify Premium is required to play tracks from BlackTape."
 *
 * 'token_expired' means: 401
 * Show message: "Spotify session expired — reconnect in Settings."
 */
export async function playTracksOnSpotify(
  trackUris: string[],
  accessToken: string
): Promise<PlayResult>;
```

Implementation details:
- getArtistTopTracks: GET https://api.spotify.com/v1/artists/${id}/top-tracks?market=from_token, returns data.tracks.map(t => t.uri)
- playTracksOnSpotify: PUT https://api.spotify.com/v1/me/player/play, body: { uris: trackUris.slice(0, 10) }, returns 'ok' on 204, 'token_expired' on 401, 'premium_required' on 403, 'no_device' on 404 or any other error
