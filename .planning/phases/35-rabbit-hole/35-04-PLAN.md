    Create the load function. Fetches artist by slug, similar artists, and streaming links. All DB access client-side (Tauri SPA — no +page.server.ts).

    ```typescript
    import type { PageLoad } from './$types';

    export const load: PageLoad = async ({ params }) => {
      try {
        const { getProvider } = await import('$lib/db/provider');
        const { getArtistBySlug, getSimilarArtists } = await import('$lib/db/queries');
        const db = await getProvider();

        const artist = await getArtistBySlug(db, params.slug);
        if (!artist) {
          return { artist: null, similarArtists: [], links: [] };
        }

        const [similarArtists, linksRaw] = await Promise.all([
          getSimilarArtists(db, artist.id, 10),
          db.all<{ platform: string; url: string }>(
            `SELECT platform, url FROM artist_links WHERE artist_id = ? ORDER BY platform`,
            artist.id
          )
        ]);

        return { artist, similarArtists, links: linksRaw };
      } catch {
        return { artist: null, similarArtists: [], links: [] };
      }
    };
    ```

    Note: `db.all` is called directly for artist_links — no wrapper function needed for a one-off query. This is consistent with how other page.ts files use `db.get` or `db.all` directly.

  <action>
    Create the artist card page. This is the main Rabbit Hole experience view.

    **Data from page load:** `artist`, `similarArtists`, `links` (via `$props().data`).

    **Async on mount:** Load releases via `get_or_cache_releases` Tauri command (non-blocking — renders without tracks first).

    **CachedRelease type** (define locally since it's a Rust command shape):
    ```typescript
    interface CachedTrack { title: string; track_number: number; length_ms: number | null; }
    interface CachedRelease { id: number; mbid: string; title: string; release_type: string; first_release_year: number | null; tracks: CachedTrack[]; }
    ```

    **Layout:**
    - Header row: artist name (link → `/artist/[slug]`) + country flag/code
    - Tags: comma-separated from `artist.tags`, each tag is a chip navigating to `/rabbit-hole/tag/[slug]`
    - Similar artists: horizontal scrollable row of chips (name only); click → navigate in-place + push trail
    - Releases section: top 3 by default; "Show all" button expands; each release shows title + year + track count
    - Action row: Play button + Continue button

    **Play button logic:**
    1. Find link matching `streamingPref.platform`
    2. If found: `shell.open(link.url)` (opens in system browser/Spotify/etc.) — external open, player bar shows "playing via X"
    3. If no preferred platform link found: try first available link (fallback)
    4. If no links at all: button disabled/dimmed

    **Continue button logic:**
    1. If `similarArtists.length > 0`: pick a random one from the list
    2. Else: call `getRandomArtistByTag(db, primaryTag, artist.id)` — primaryTag = first tag from `artist.tags.split(',')[0].trim()`
    3. Navigate to the result's slug, push to trail

    **Tag chip navigation:** Use `goto('/rabbit-hole/tag/' + encodeURIComponent(tag), { keepFocus: true, noScroll: true })` + pushTrailItem. Do NOT reuse the existing `TagChip.svelte` component — it hardcodes `/search` links (per research pitfall 5).

    ```svelte
    <script lang="ts">
      import { goto } from '$app/navigation';
      import { onMount } from 'svelte';
      import { isTauri } from '$lib/platform';
      import { streamingPref } from '$lib/theme/preferences.svelte';
      import { pushTrailItem } from '$lib/rabbit-hole/trail.svelte';
      import { getProvider } from '$lib/db/provider';
      import { getRandomArtistByTag } from '$lib/db/queries';

      interface CachedTrack { title: string; track_number: number; length_ms: number | null; }
      interface CachedRelease { id: number; mbid: string; title: string; release_type: string; first_release_year: number | null; tracks: CachedTrack[]; }

      let { data } = $props();
      let { artist, similarArtists, links } = $derived(data);

      let releases = $state<CachedRelease[]>([]);
      let releasesLoading = $state(false);
      let showAllReleases = $state(false);
      let continueLoading = $state(false);

      const TOP_RELEASES = 3;
      let visibleReleases = $derived(showAllReleases ? releases : releases.slice(0, TOP_RELEASES));

      onMount(async () => {
        if (!artist || !isTauri()) return;
        releasesLoading = true;
        try {
          const { invoke } = await import('@tauri-apps/api/core');
          releases = await invoke<CachedRelease[]>('get_or_cache_releases', {
            artistMbid: artist.mbid
          });
        } catch {
          // best-effort — no tracks available
        } finally {
          releasesLoading = false;
        }
      });

      function navigateToArtist(slug: string, name: string) {
        pushTrailItem({ type: 'artist', slug, name });
        goto(`/rabbit-hole/artist/${slug}`, { keepFocus: true, noScroll: true });
      }

      function navigateToTag(tag: string) {
        const slug = encodeURIComponent(tag.trim());
        pushTrailItem({ type: 'tag', slug, name: tag.trim() });
        goto(`/rabbit-hole/tag/${slug}`, { keepFocus: true, noScroll: true });
      }

      async function handlePlay() {
        if (!links || links.length === 0) return;
        const pref = streamingPref.platform;
        const preferred = links.find((l: { platform: string; url: string }) => l.platform === pref);
        const link = preferred ?? links[0];
        if (!link) return;
        try {
          const { open } = await import('@tauri-apps/plugin-shell');
          await open(link.url);
        } catch { /* ignore */ }
      }

      async function handleContinue() {
        if (!artist || continueLoading) return;
        continueLoading = true;
        try {
          if (similarArtists.length > 0) {
            const pick = similarArtists[Math.floor(Math.random() * similarArtists.length)];
            navigateToArtist(pick.slug, pick.name);
          } else {
            // Fallback: random artist sharing primary tag
            const primaryTag = (artist.tags ?? '').split(',')[0]?.trim();
            if (!primaryTag) return;
            const db = await getProvider();
            const fallback = await getRandomArtistByTag(db, primaryTag, artist.id);
            if (fallback) {
              navigateToArtist(fallback.slug, fallback.name);
            }
          }
        } catch { /* ignore */ } finally {
          continueLoading = false;
        }
      }

      function formatDuration(ms: number | null): string {
        if (!ms) return '';
        const s = Math.floor(ms / 1000);
        const m = Math.floor(s / 60);
        const sec = s % 60;
        return `${m}:${sec.toString().padStart(2, '0')}`;
      }

      let tagList = $derived(
        artist?.tags ? artist.tags.split(',').map((t: string) => t.trim()).filter(Boolean) : []
      );
      let hasLinks = $derived(links && links.length > 0);
    </script>

    {#if !artist}
      <div class="rh-card-wrap">
        <div class="rh-not-found">Artist not found.</div>
      </div>
    {:else}
      <div class="rh-card-wrap">
        <div class="rh-card">

          <!-- Header -->

            <a href="/artist/{artist.slug}" class="rh-artist-name">{artist.name}</a>
            {#if artist.country}

          <!-- Tags -->

            </div>
          {/if}

                {#each similarArtists as sa}

                    {sa.name}
                  </button>
                {/each}
              </div>
            </div>
          {/if}

          <!-- Releases / Tracks -->
          <div class="rh-section">
            <div class="rh-section-label">Releases</div>
            {#if releasesLoading}
              <div class="rh-loading-hint">Loading releases...</div>
            {:else if releases.length === 0}
              <div class="rh-loading-hint">No releases cached yet.</div>
            {:else}
              <div class="rh-releases">
                {#each visibleReleases as rel}
                  <div class="rh-release">
                    <div class="rh-release-header">
                      <span class="rh-release-title">{rel.title}</span>
                      <span class="rh-release-meta">
                        {rel.first_release_year ?? ''}
                        {#if rel.tracks.length > 0}· {rel.tracks.length} tracks{/if}
                      </span>
                    </div>
                    {#if rel.tracks.length > 0}
                      <div class="rh-tracks">
                        {#each rel.tracks.slice(0, 5) as track}
                          <div class="rh-track">
                            <span class="rh-track-num">{track.track_number}</span>
                            <span class="rh-track-title">{track.title}</span>
                            {#if track.length_ms}

                            {/if}
                          </div>
                        {/each}
                        {#if rel.tracks.length > 5}
                          <div class="rh-track-more">+{rel.tracks.length - 5} more tracks</div>
                        {/if}
                      </div>
                    {/if}
                  </div>
                {/each}
              </div>
              {#if releases.length > TOP_RELEASES}

                </button>
              {/if}
            {/if}
          </div>

          <!-- Actions -->
          <div class="rh-actions">
            <button
              class="rh-play-btn"

              ▶ Play

          </div>
    ```
